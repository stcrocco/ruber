def inspect
  return "<#{self.class}:#{object_id} DISPOSED>" if disposed?
  start_c = self.start
  end_c = self.end
  "<#{self.class}:#{object_id} start=(#{start_c.line};#{start_c.column}) end=(#{end_c.line};#{end_c.column})>"
end
