def format
  res = ""
  res << KDE.i18n("Line %d") % (line + 1) if line
  res << KDE.i18n(", column %d") % (column + 1) if line and column
  msg = formatted_message || message
  res << ": " if msg and !res.empty?
  res << msg
  res
end
