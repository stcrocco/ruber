def initialize parent = nil
  super

  @ui.setup_ui self

  connect @ui.search, SIGNAL(:clicked), self, SLOT(:start_search)
  @ui.content.open_links = false
  connect @ui.content, SIGNAL('anchorClicked(QUrl)'), Ruber[:ruberri], SLOT('display_url(QUrl)')
end

  unless text.empty?

  end
