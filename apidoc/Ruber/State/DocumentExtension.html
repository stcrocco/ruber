Overview

Extension for documents needed by the State plugin

The scope of this extension is to save and restore the position of the cursor in the document

def restore
  view = @document.view
  return unless view
  pos = @document.own_project[:state, :cursor_position]
  view.go_to *pos
  nil
end

def save_settings
  view = @document.view 
  pos = if view
    cur = view.cursor_position
    [cur.line, cur.column]
  else [0,0]
  end
  @project[:state, :cursor_position] = pos
  nil
end
